  <script>
    /* =========================================================
       JS DO LAYOUT
       - Sidebar desktop retrátil
       - Sidebar mobile drawer
       - Submenus acessíveis
       - Light/Dark mode
       - Color switcher
       ========================================================= */

    const appRoot = document.getElementById('appRoot');
    const sidebar = document.getElementById('sidebar');
    const overlay = document.getElementById('overlay');
    const sidebarToggle = document.getElementById('sidebarToggle');
    const themeIcon = document.getElementById('themeIcon');
    const primaryColorSelect = document.getElementById('primaryColorSelect');

    const desktopQuery = window.matchMedia('(min-width: 1024px)');

    const colorThemes = {
      blue: {
        primary: '#2563eb',
        hover: '#1d4ed8',
        light: 'rgba(37, 99, 235, 0.11)'
      },
      green: {
        primary: '#059669',
        hover: '#047857',
        light: 'rgba(5, 150, 105, 0.12)'
      },
      purple: {
        primary: '#7c3aed',
        hover: '#6d28d9',
        light: 'rgba(124, 58, 237, 0.12)'
      },
      orange: {
        primary: '#ea580c',
        hover: '#c2410c',
        light: 'rgba(234, 88, 12, 0.12)'
      },
      slate: {
        primary: '#334155',
        hover: '#1e293b',
        light: 'rgba(51, 65, 85, 0.12)'
      }
    };

    function isDesktop() {
      return desktopQuery.matches;
    }

    function setSidebarToggleState() {
      const isMobileOpen = sidebar.classList.contains('active');
      const isDesktopCollapsed = appRoot.classList.contains('sidebar-collapsed');

      if (isDesktop()) {
        sidebarToggle.setAttribute('aria-expanded', String(!isDesktopCollapsed));
      } else {
        sidebarToggle.setAttribute('aria-expanded', String(isMobileOpen));
      }
    }

    function openMobileSidebar() {
      sidebar.classList.add('active');
      overlay.classList.add('active');
      document.body.style.overflow = 'hidden';
      setSidebarToggleState();
    }

    function closeMobileSidebar() {
      sidebar.classList.remove('active');
      overlay.classList.remove('active');
      document.body.style.overflow = '';
      setSidebarToggleState();
    }

    function toggleSidebar() {
      if (isDesktop()) {
        appRoot.classList.toggle('sidebar-collapsed');
        localStorage.setItem(
          'controleTI.sidebarCollapsed',
          appRoot.classList.contains('sidebar-collapsed') ? 'true' : 'false'
        );
      } else {
        if (sidebar.classList.contains('active')) {
          closeMobileSidebar();
        } else {
          openMobileSidebar();
        }
      }

      setSidebarToggleState();
    }

    function toggleMenu(id, element) {
      const submenu = document.getElementById(id);

      if (!submenu) return;

      if (isDesktop() && appRoot.classList.contains('sidebar-collapsed')) {
        appRoot.classList.remove('sidebar-collapsed');
        localStorage.setItem('controleTI.sidebarCollapsed', 'false');
        setSidebarToggleState();
      }

      const isOpen = submenu.classList.contains('open');

      document.querySelectorAll('.submenu').forEach((item) => {
        if (item.id !== id) {
          item.classList.remove('open');
        }
      });

      document.querySelectorAll('.menu-trigger').forEach((trigger) => {
        if (trigger !== element) {
          trigger.setAttribute('aria-expanded', 'false');
        }
      });

      submenu.classList.toggle('open', !isOpen);
      element.setAttribute('aria-expanded', String(!isOpen));
    }

    function applyTheme(theme) {
      const selectedTheme = theme === 'dark' ? 'dark' : 'light';

      document.body.classList.toggle('dark-mode', selectedTheme === 'dark');
      document.body.classList.toggle('light-mode', selectedTheme === 'light');

      localStorage.setItem('controleTI.theme', selectedTheme);

      if (themeIcon) {
        themeIcon.className = selectedTheme === 'dark' ? 'fas fa-sun' : 'fas fa-moon';
      }
    }

    function toggleTheme() {
      const isDark = document.body.classList.contains('dark-mode');
      applyTheme(isDark ? 'light' : 'dark');
    }

    function applyPrimaryColor(colorName) {
      const selected = colorThemes[colorName] ? colorName : 'blue';
      const color = colorThemes[selected];

      document.documentElement.style.setProperty('--primary', color.primary);
      document.documentElement.style.setProperty('--primary-hover', color.hover);
      document.documentElement.style.setProperty('--primary-light', color.light);

      localStorage.setItem('controleTI.primaryColor', selected);

      if (primaryColorSelect) {
        primaryColorSelect.value = selected;
      }
    }

    function openActiveSubmenu() {
      document.querySelectorAll('.sub-link.active').forEach((link) => {
        const submenu = link.closest('.submenu');

        if (!submenu) return;

        submenu.classList.add('open');

        const trigger = document.querySelector(`[aria-controls="${submenu.id}"]`);

        if (trigger) {
          trigger.classList.add('active');
          trigger.setAttribute('aria-expanded', 'true');
        }
      });
    }

    function closeSidebarOnMobileNavigation() {
      document.querySelectorAll('.sidebar a').forEach((link) => {
        link.addEventListener('click', () => {
          if (!isDesktop()) {
            closeMobileSidebar();
          }
        });
      });
    }

    function restoreLayoutPreferences() {
      const savedTheme = localStorage.getItem('controleTI.theme');
      const savedColor = localStorage.getItem('controleTI.primaryColor');
      const savedSidebar = localStorage.getItem('controleTI.sidebarCollapsed');

      applyTheme(savedTheme || 'light');
      applyPrimaryColor(savedColor || 'blue');

      if (savedSidebar === 'true' && isDesktop()) {
        appRoot.classList.add('sidebar-collapsed');
      }

      setSidebarToggleState();
    }

    document.addEventListener('DOMContentLoaded', () => {
      restoreLayoutPreferences();
      openActiveSubmenu();
      closeSidebarOnMobileNavigation();

      if (primaryColorSelect) {
        primaryColorSelect.addEventListener('change', (event) => {
          applyPrimaryColor(event.target.value);
        });
      }
    });

    document.addEventListener('keydown', (event) => {
      if (event.key === 'Escape' && sidebar.classList.contains('active')) {
        closeMobileSidebar();
      }
    });

    desktopQuery.addEventListener('change', () => {
      closeMobileSidebar();

      const savedSidebar = localStorage.getItem('controleTI.sidebarCollapsed');

      if (isDesktop() && savedSidebar === 'true') {
        appRoot.classList.add('sidebar-collapsed');
      } else if (!isDesktop()) {
        appRoot.classList.remove('sidebar-collapsed');
      }

      setSidebarToggleState();
    });
  </script>
