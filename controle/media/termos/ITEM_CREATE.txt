@login_required
@transaction.atomic  # Garante que tudo ou nada seja salvo 
def item_create(request):
    if request.method == "POST":
        form = ItemForm(request.POST)
        locacao_form = LocacaoForm(request.POST)
        lote_form = LoteEstoqueForm(request.POST)

        form_valido = form.is_valid()

        tem_lote = False
        eh_locado = False

        if form_valido:
            tem_lote = form.cleaned_data.get("tem_lote") is True
            eh_locado = form.cleaned_data.get("locado") == SimNaoChoices.SIM

        locacao_valida = locacao_form.is_valid() if eh_locado else True
        lote_valido = lote_form.is_valid() if tem_lote else True

        if form_valido and locacao_valida and lote_valido:
            try:
                with transaction.atomic():
                    item = form.save(commit=False)
                    preencher_auditoria(item, request.user, criando=True)

                    lote = None

                    if tem_lote:
                        lote = lote_form.save(commit=False)
                        lote.quantidade = item.quantidade or 1

                        preencher_auditoria(lote, request.user, criando=True)

                        # Importante:
                        # full_clean depois de preencher auditoria.
                        lote.full_clean()
                        lote.save()

                        item.fornecedor = lote.fornecedor
                        item.valor = lote.custo_unitario
                        item.numero_pedido = lote.numero_nf

                    preencher_auditoria(item, request.user, criando=True)
                    item.save()

                    if tem_lote and lote:
                        item_lote = ItemLote(
                            item=item,
                            lote=lote,
                            quantidade_entrada=lote.quantidade,
                            quantidade_disponivel=lote.quantidade,
                            custo_unitario=lote.custo_unitario,
                            custo_total=lote.custo_total,
                        )

                        preencher_auditoria(item_lote, request.user, criando=True)
                        item_lote.full_clean()
                        item_lote.save()

                    if eh_locado:
                        locacao = locacao_form.save(commit=False)
                        locacao.equipamento = item

                        preencher_auditoria(locacao, request.user, criando=True)
                        locacao.full_clean()
                        locacao.save()

                    messages.success(request, "Equipamento cadastrado com sucesso.")
                    return redirect("equipamentos_list")

            except ValidationError as e:
                adicionar_erros_validacao_no_form(form, e)
                messages.error(request, "Erro de validação. Verifique os dados informados.")

            except IntegrityError as e:
                messages.error(
                    request,
                    f"Erro de integridade ao salvar o equipamento. Nenhum dado foi gravado. Detalhe: {str(e)}"
                )

            except Exception as e:
                messages.error(
                    request,
                    f"Erro inesperado ao salvar o equipamento: {str(e)}"
                )

        else:
            messages.error(request, "Verifique os campos obrigatórios antes de salvar.")

    else:
        form = ItemForm()
        locacao_form = LocacaoForm()
        lote_form = LoteEstoqueForm()

    return render(request, "front/equipamentos/cadastrar_equipamento.html", {
        "form": form,
        "locacao_form": locacao_form,
        "lote_form": lote_form,
    })

<script>
document.addEventListener('DOMContentLoaded', function() {
    
    // 1. Aplica estilo aos inputs padrões do Django (caso widget não tenha classe)
    const inputs = document.querySelectorAll('input:not([type=checkbox]):not([type=radio]), textarea, select');
    inputs.forEach(el => el.classList.add('form-control'));

    // 2. Inicializa Choices.js (Select2 style) automaticamente em todos os selects
    const selects = document.querySelectorAll('select');
    selects.forEach(select => {
        new Choices(select, {
            searchEnabled: true,
            searchPlaceholderValue: 'Pesquisar...',
            itemSelectText: '',
            shouldSort: false, // Mantém ordem do Django
            placeholder: true,
            allowHTML: false
        });
    });

    // 3. Lógica de Mostrar/Ocultar Locação
    const fieldLocado = document.getElementById('{{ form.locado.auto_id }}');
    const boxLocado = document.getElementById('box_locado');
    const boxNaoLocado = document.getElementById('box_nao_locado');

    function toggleLocacao() {
        if (!fieldLocado) return;
        
        // Verifica se é "Sim" (compatível com ChoiceField 'sim', 'nao' ou Boolean)
        const val = fieldLocado.value.toLowerCase();
        const isLocado = (val === 'sim' || val === 'true' || val === '1');

        if (isLocado) {
            boxLocado.style.display = 'block';
            boxNaoLocado.style.display = 'none';
        } else {
            boxLocado.style.display = 'none';
            boxNaoLocado.style.display = 'block';
        }
    }

    if (fieldLocado) {
        fieldLocado.addEventListener('change', toggleLocacao);
        // Dispara no load para pegar o estado inicial (edição)
        toggleLocacao();
    }

    // 4. Lógica de Preventiva
    const fieldPrev = document.getElementById('{{ form.precisa_preventiva.auto_id }}');
    const grpPrev = document.getElementById('grp_preventiva');

    function togglePreventiva() {
        if (!fieldPrev) return;
        const val = fieldPrev.value.toLowerCase();
        const needsPrev = (val === 'sim' || val === 'true' || val === '1');
        
        // Exibe o grupo se necessário. 'flex' porque é um container de grid/flex no CSS
        if (grpPrev) {
            grpPrev.style.display = needsPrev ? 'flex' : 'none'; 
        }
    }

    if (fieldPrev) {
        fieldPrev.addEventListener('change', togglePreventiva);
        togglePreventiva();
    }
    const fieldTemLote = document.getElementById('{{ form.tem_lote.auto_id }}');
    const boxLote = document.getElementById('box_lote');

    function toggleLote() {
        if (!fieldTemLote || !boxLote) return;

        let temLote = false;

        if (fieldTemLote.type === 'checkbox') {
            temLote = fieldTemLote.checked;
        } else {
            const val = String(fieldTemLote.value).toLowerCase();
            temLote = (val === 'sim' || val === 'true' || val === '1');

class LoteEstoque(AuditModel):
    fornecedor = models.ForeignKey(
        "Fornecedor",
        on_delete=models.PROTECT,
        related_name="lotes_estoque",
        verbose_name="Fornecedor"
    )

    data_entrada = models.DateField(
        verbose_name="Data de Entrada"
    )

    numero_nf = models.CharField(
        max_length=60,
        verbose_name="Número da NF"
    )

    serie_nf = models.CharField(
        max_length=30,
        verbose_name="Série da NF"
    )

    quantidade = models.PositiveIntegerField(
        default=1,
        validators=[MinValueValidator(1)],
        verbose_name="Quantidade do Lote"
    )

    custo_unitario = models.DecimalField(
        max_digits=12,
        decimal_places=2,
        validators=[MinValueValidator(Decimal("0.01"))],
        verbose_name="Custo Unitário"
    )

    custo_total = models.DecimalField(
        max_digits=12,
        decimal_places=2,
        validators=[MinValueValidator(Decimal("0.01"))],
        verbose_name="Custo Total"
    )

    observacao_tecnica = models.TextField(
        blank=True,
        null=True,
        verbose_name="Observação Técnica"
    )

    class Meta:
        verbose_name = "Lote de Estoque"
        verbose_name_plural = "Lotes de Estoque"
        ordering = ["-data_entrada", "-created_at"]
        indexes = [
            models.Index(fields=["numero_nf", "serie_nf"]),
            models.Index(fields=["data_entrada"]),
        ]

    def clean(self):
        super().clean()

        if not self.fornecedor_id:
            raise ValidationError({"fornecedor": "Fornecedor é obrigatório para o lote."})

        if not self.data_entrada:
            raise ValidationError({"data_entrada": "Data de entrada é obrigatória para o lote."})

        if not self.numero_nf:
            raise ValidationError({"numero_nf": "Número da NF é obrigatório para o lote."})

        if not self.serie_nf:
            raise ValidationError({"serie_nf": "Série da NF é obrigatória para o lote."})

        if not self.quantidade or self.quantidade <= 0:
            raise ValidationError({"quantidade": "A quantidade do lote deve ser maior que zero."})

        if not self.custo_unitario or self.custo_unitario <= 0:
            raise ValidationError({"custo_unitario": "O custo unitário deve ser maior que zero."})

        if not self.custo_total or self.custo_total <= 0:
            raise ValidationError({"custo_total": "O custo total deve ser maior que zero."})

        total_calculado = self.custo_unitario * Decimal(self.quantidade)

        if self.custo_total != total_calculado:
            raise ValidationError({
                "custo_total": (
                    f"Custo total inconsistente. Esperado: R$ {total_calculado:.2f} "
                    f"com base em {self.quantidade} unidade(s) x R$ {self.custo_unitario:.2f}."
                )
            })

    def __str__(self):
        return f"Lote NF {self.numero_nf}/{self.serie_nf} - {self.fornecedor}"

        verbose_name="Lote"
    )

    quantidade_entrada = models.PositiveIntegerField(
        validators=[MinValueValidator(1)],
        verbose_name="Quantidade de Entrada"
    )

    quantidade_disponivel = models.PositiveIntegerField(
        validators=[MinValueValidator(0)],
        verbose_name="Quantidade Disponível"
    )

    custo_unitario = models.DecimalField(
        max_digits=12,
        decimal_places=2,
        validators=[MinValueValidator(Decimal("0.01"))],
        verbose_name="Custo Unitário"
    )

    custo_total = models.DecimalField(
        max_digits=12,
        decimal_places=2,
        validators=[MinValueValidator(Decimal("0.01"))],
        verbose_name="Custo Total"
    )

    class Meta:
        verbose_name = "Vínculo Item x Lote"
        verbose_name_plural = "Vínculos Item x Lote"
        ordering = ["-created_at"]

    def clean(self):
        super().clean()

        if not self.item_id:
            raise ValidationError({"item": "Item é obrigatório no vínculo de lote."})

        if not self.lote_id:
            raise ValidationError({"lote": "Lote é obrigatório."})

        if self.quantidade_disponivel > self.quantidade_entrada:
            raise ValidationError({
                "quantidade_disponivel": "A quantidade disponível não pode ser maior que a quantidade de entrada."
            })

        return f"{self.item} | {self.lote} | Qtd: {self.quantidade_entrada}"
