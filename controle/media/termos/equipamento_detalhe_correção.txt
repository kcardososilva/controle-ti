@login_required
def equipamento_detalhe(request, pk: int):
    """
    Exibe os detalhes completos de um ativo, incluindo status calculado de preventivas
    e histórico de movimentações.
    """
    # 1. Busca eficiente do item com todas as relações necessárias (JOINs)
    item = get_object_or_404(
        Item.objects.select_related(
            "subtipo", 
            "localidade", 
            "centro_custo", 
            "fornecedor"
        ),
        pk=pk,
    )

    # 2. Histórico de Movimentações (Ordenado do mais recente)
    movimentacoes = (
        MovimentacaoItem.objects
        .filter(item=item)
        .select_related(
            "usuario",
            "localidade_destino", "centro_custo_destino",
            "localidade_origem", "centro_custo_origem",
        )
        .order_by("-created_at")
    )

    # 3. Preventivas e Cálculo de Datas
    preventivas = (
        Preventiva.objects
        .filter(equipamento=item)
        .select_related("checklist_modelo")
        .order_by("-data_proxima", "-data_ultima")
    )

    today = timezone.localdate()

    for p in preventivas:
        # Se não tiver data próxima definida no banco, tentamos calcular em tempo real
        if not p.data_proxima:
            intervalo = 0
            
            # Tenta pegar intervalo do modelo de checklist
            if p.checklist_modelo and p.checklist_modelo.intervalo_dias:
                try:
                    intervalo = int(p.checklist_modelo.intervalo_dias)
                except (ValueError, TypeError):
                    intervalo = 0
            
            # Se falhar, tenta pegar do cadastro do item
            elif item.data_limite_preventiva:
                try:
                    intervalo = int(item.data_limite_preventiva)
                except (ValueError, TypeError):
                    intervalo = 0

            # Se encontrou um intervalo válido, projeta a data
            if intervalo > 0:
                base_date = p.data_ultima if p.data_ultima else today
                p.data_proxima = base_date + timedelta(days=intervalo)

        # Flag booleana para facilitar estilização no template
        # Está em dia se não tem data definida (null) OU se a data futura ainda não chegou
        p.flag_em_dia = (p.data_proxima is None) or (p.data_proxima >= today)

    # 4. Dados Opcionais (Try/Except para evitar quebra se o model não existir ou mudar)
    comentarios = []
    try:
        # Verifica se o model existe no contexto do projeto
        from .models import ComentarioEquipamento 
        comentarios = ComentarioEquipamento.objects.filter(
            equipamento=item
        ).select_related("criado_por").order_by("-created_at")
    except ImportError:
        pass # Model não existe, lista vazia
    except Exception:
        pass # Erro genérico de DB

    locacao = None
    try:
        # Tenta acessar relação OneToOne ou Reverse FK
        if hasattr(item, 'locacao'):
            locacao = item.locacao
    except Exception:
        pass

    context = {
        "item": item,
        "movimentacoes": movimentacoes,
        "preventivas": preventivas,
        "comentarios": comentarios,
        "locacao": locacao,
        "today": today,
    }

    return render(request, "front/equipamentos/equipamento_detalhe.html", context)
